from fastapi import APIRouter, Depends, HTTPException

@router.post("/nova/start-session")
async def start_nova_session(
    user_id: UUID,
    context_type: str,  # 'onboarding', 'sphere_setup', 'tutorial'
    sphere_id: Optional[str] = None,
    db: AsyncSession = Depends(get_db)
):
    """
    Démarre une nouvelle session de dialogue avec Nova.
    """
    session_id = await create_nova_session(db, user_id, context_type, sphere_id)
    
    # Premier message de Nova
    initial_message = await get_nova_greeting(context_type, sphere_id)
    
    await save_nova_message(db, session_id, 'nova', initial_message)
    
    return {
        "session_id": session_id,
        "nova_message": initial_message,
        "next_action": "await_user_input"
    }

@router.post("/nova/dialogue")
async def nova_dialogue(
    user_id: UUID,
    request: NovaDialogueRequest,
    db: AsyncSession = Depends(get_db)
):
    """
    Continue le dialogue avec Nova.
    Nova analyse la réponse et génère la suite.
    """
    # Sauvegarder le message utilisateur
    await save_nova_message(
        db, request.session_id, 'user', 
        request.user_message, request.current_step
    )
    
    # Récupérer le contexte du dialogue
    context = await get_dialogue_context(db, request.session_id)
    
    # Générer la réponse Nova
    nova_response = await generate_nova_response(
        db, user_id, context, request.user_message, request.sphere_id
    )
    
    # Sauvegarder la réponse Nova
    await save_nova_message(
        db, request.session_id, 'nova',
        nova_response['message'],
        nova_response.get('message_type'),
        nova_response.get('actions'),
        nova_response.get('structured_data')
    )
    
    return {
        "session_id": request.session_id,
        "nova_message": nova_response['message'],
        "message_type": nova_response.get('message_type'),
        "actions": nova_response.get('actions'),
        "suggestions": nova_response.get('suggestions'),
        "next_step": nova_response.get('next_step')
    }
