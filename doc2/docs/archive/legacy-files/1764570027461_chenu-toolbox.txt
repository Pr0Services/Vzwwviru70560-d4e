const ChenuToolbox = ({ isOpen, onClose, onSelectTool, onConfigureTool, assignedTools = [] }) => {
  const [activeCategory, setActiveCategory] = useState('all');
  const [searchQuery, setSearchQuery] = useState('');
  const [selectedTool, setSelectedTool] = useState(null);
  const [viewMode, setViewMode] = useState('grid');

  const filteredTools = TOOLS.filter(tool => {
    const matchesCategory = activeCategory === 'all' || tool.category === activeCategory;
    const matchesSearch = tool.name.toLowerCase().includes(searchQuery.toLowerCase()) || 
                          tool.description.toLowerCase().includes(searchQuery.toLowerCase());
    return matchesCategory && matchesSearch;
  });

  const toolsByCategory = filteredTools.reduce((acc, tool) => {
    if (!acc[tool.category]) acc[tool.category] = [];
    acc[tool.category].push(tool);
    return acc;
  }, {});

  const stats = {
    total: TOOLS.length,
    assigned: assignedTools.length,
    byCategory: Object.entries(TOOL_CATEGORIES).map(([id, cat]) => ({
      ...cat,
      count: TOOLS.filter(t => t.category === id).length
    }))
  };

  if (!isOpen) return null;

  return (
    <div className="tb-overlay">
      <div className="tb-modal">
        <header className="tb-header">
          <div className="tb-header-left">
            <span>üß∞</span>
            <div><h1>Toolbox</h1><p>{stats.total} tools available</p></div>
          </div>
          <div className="tb-header-right">
            <div className="tb-search">
              <span>üîç</span>
              <input value={searchQuery} onChange={e => setSearchQuery(e.target.value)} placeholder="Search tools..." />
            </div>
            <div className="tb-view-toggle">
              <button className={viewMode === 'grid' ? 'active' : ''} onClick={() => setViewMode('grid')}>‚ñ¶</button>
              <button className={viewMode === 'list' ? 'active' : ''} onClick={() => setViewMode('list')}>‚ò∞</button>
            </div>
            <button className="tb-close" onClick={onClose}>‚úï</button>
          </div>
        </header>

        <div className="tb-body">
          <aside className="tb-sidebar">
            <div className="tb-categories">
              <button className={`tb-cat-btn ${activeCategory === 'all' ? 'active' : ''}`} onClick={() => setActiveCategory('all')}>
                <span>üìö</span><span>All Tools</span><span className="tb-count">{stats.total}</span>
              </button>
              {stats.byCategory.map(cat => (
                <button key={cat.id} className={`tb-cat-btn ${activeCategory === cat.id ? 'active' : ''}`} onClick={() => setActiveCategory(cat.id)} style={{'--cat-color': cat.color}}>
                  <span>{cat.icon}</span><span>{cat.name}</span><span className="tb-count">{cat.count}</span>
                </button>
              ))}
            </div>
          </aside>

          <main className="tb-main">
            {viewMode === 'grid' ? (
              <div className="tb-tools-grid">
                {filteredTools.map(tool => {
                  const category = TOOL_CATEGORIES[tool.category];
                  const isAssigned = assignedTools.includes(tool.id);
                  return (
                    <div key={tool.id} className={`tb-tool-card ${selectedTool?.id === tool.id ? 'selected' : ''} ${isAssigned ? 'assigned' : ''}`} onClick={() => setSelectedTool(tool)} style={{'--tool-color': category?.color}}>
                      <div className="tb-tool-header">
                        <span className="tb-tool-icon">{tool.icon}</span>
                        {isAssigned && <span className="tb-assigned-badge">‚úì</span>}
                      </div>
                      <h3>{tool.name}</h3>
                      <p>{tool.description}</p>
                      <div className="tb-tool-meta">
                        <span className="tb-tool-category">{category?.icon} {category?.name}</span>
                        <span className="tb-tool-cost">ü™ô {tool.tokenCost}</span>
                      </div>
                    </div>
                  );
                })}
              </div>
            ) : (
              <div className="tb-tools-list">
                {Object.entries(toolsByCategory).map(([catId, tools]) => {
                  const category = TOOL_CATEGORIES[catId];
                  return (
                    <div key={catId} className="tb-list-section">
                      <h3 style={{color: category?.color}}>{category?.icon} {category?.name}</h3>
                      {tools.map(tool => (
                        <div key={tool.id} className={`tb-list-item ${selectedTool?.id === tool.id ? 'selected' : ''}`} onClick={() => setSelectedTool(tool)}>
                          <span className="tb-list-icon">{tool.icon}</span>
                          <div className="tb-list-info"><strong>{tool.name}</strong><small>{tool.description}</small></div>
                          <span className="tb-list-cost">ü™ô {tool.tokenCost}</span>
                          <button className="tb-list-add" onClick={(e) => { e.stopPropagation(); onSelectTool?.(tool); }}>+</button>
                        </div>
                      ))}
                    </div>
                  );
                })}
              </div>
            )}
          </main>

          {selectedTool && (
            <aside className="tb-detail">
              <div className="tb-detail-header" style={{background: `${TOOL_CATEGORIES[selectedTool.category]?.color}20`}}>
                <span className="tb-detail-icon">{selectedTool.icon}</span>
                <div><h2>{selectedTool.name}</h2><span className="tb-detail-cat">{TOOL_CATEGORIES[selectedTool.category]?.name}</span></div>
              </div>
              <div className="tb-detail-body">
                <p className="tb-detail-desc">{selectedTool.description}</p>
                
                <div className="tb-detail-section">
                  <h4>üì• Inputs</h4>
                  <div className="tb-io-list">{selectedTool.inputs.map((input, i) => <span key={i} className="tb-io-tag input">{input}</span>)}</div>
                </div>
                
                <div className="tb-detail-section">
                  <h4>üì§ Outputs</h4>
                  <div className="tb-io-list">{selectedTool.outputs.map((output, i) => <span key={i} className="tb-io-tag output">{output}</span>)}</div>
                </div>
                
                <div className="tb-detail-section">
                  <h4>ü™ô Token Cost</h4>
                  <div className="tb-cost-display">{selectedTool.tokenCost} tokens/call</div>
                </div>
              </div>
              <div className="tb-detail-actions">
                <button className="tb-btn-configure" onClick={() => onConfigureTool?.(selectedTool)}>‚öôÔ∏è Configure</button>
                <button className="tb-btn-add" onClick={() => { onSelectTool?.(selectedTool); }}>‚ûï Add to Agent</button>
              </div>
            </aside>
          )}
        </div>
      </div>
    </div>
  );
};

export default ChenuToolbox;
